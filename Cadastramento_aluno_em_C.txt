#include <stdio.h>
#include <stdlib.h>
#include <string.h>


#define tamanho 200
char alunoNome[50];
char email[50];
int cpf[50];
int dataP[50];
int dataF[50];
float horaP[50];
float horaF[50];
int op;

void pesquisa();
void cadastro();
void lista();
int main (void){
  do{

    printf("\n Menu");
    printf("\n1: cadastrar");
    printf("\n2: listar");
    printf("\n3: pesquisa");
    printf("\n4: sair \n");
    scanf("%d", &op);
    
    switch (op){
      case 1:
      cadastro();

      break;
      
      case 2:
      lista();
      break;

      case 3:
    pesquisa();
      break;

      case 4:
      system("Exit");
      break;

      default:
      printf("opção inválida");
      break;


    }

  }while(op != 4);
  
}
void lista (){
int i;

for(i = 0; i<tamanho;i++){
if(cpf[i]>0){
  printf("\nNome: %c\nEmail: %c\nCPF: %d", alunoNome[i], email[i], cpf[i]);
  printf("\nData da Presença: %d", dataP[i]);
  printf("\nHora da Presença: %1.f", horaP[i]);
  printf("\nData da Falta: %d", dataF[i]);
  printf("\nHora da Falta: %1.f",horaF[i]);
}else{
  break;
}


}


}

void cadastro (){
static int linha;
do{
printf("\n Digite seu nome: ");
scanf("%s",&alunoNome[linha]);
printf("\n Digite sua cpf: ");
scanf("%d",&cpf[linha]);
printf("\n Digite seu email: ");
scanf("%s",&email[linha]);
printf("\n Digite o dia que estava presente na aula: ");
scanf("%d",&dataP[linha]);
printf("\n Digite a hora que estava presente na aula: ");
scanf("%f",&horaP[linha]);
printf("\n Digite o dia que faltou: ");
scanf("%d",&dataF[linha]);
printf("\n Digite a hora que faltou: ");
scanf("%f",&horaF[linha]);

printf("\n Digite 1 para continuar ou outro valor para sair: ");
scanf("%d",&op);
linha++;

}while(op == 1);

}//cadastro

void pesquisa(){
  int cpfPesquisa, i;
  char emailPesquisa;

  do{
    printf("\n Digite 1 para pesquisar por data de nascimento: ");
    scanf("%d",&op);
    switch(op){
      case 1:
      printf("\nDigite o seu cpf: ");
      scanf("%d", &cpfPesquisa);
      for(i=0; i<tamanho;i++){
        if(cpf[i]==cpfPesquisa){
          printf("\n Nome: %c\nEmail: %d\nCPF: %d",alunoNome[i], email[i], cpf[i]);
        }
      }
      break;

      case 2:
      printf("\n Digite o seu email: ");
      scanf("%s", &emailPesquisa);
      for(i=0; i<tamanho;i++){
        if(email[i]==emailPesquisa){
          printf("\nNome: %c\nEmail: %d\nCPF: %d", alunoNome[i], email[i], cpf[i]);
        }

      }
      break;
      default:
      printf("Opção invalida");
      break;
    }

    printf("Digite 1 para continuar pesquisando");
    scanf("%d", &op);

  }while (op == 1);
}//Pesquisa
